            n++;
            if (n === $(".imgbox li").length) {
                n = 0;
            }
            if (n < 0) {
                n = $(".imgbox li").length - 1;
            }
            $(".imgbox li").each(function () {
                flag = true;
                $(".imgbox li").removeClass("imgbox1").eq(n).addClass("imgbox1");
                $(".pagerbox li").removeClass("active").eq(n).addClass("active");
            });
        }
        $(".btn_right").click(function () {
            if (flag) {
                flag = false;
                move();
            }
        });
        $(".btn_left").click(function () {
            if (flag) {
                flag = false;
                n -= 2;
                move();
            }
        });
    }

        let left = $(".star_btn1");
        let right = $(".star_btn2 ");

            n++;

            if (n == 2) {
                right.attr("disable", false);
            }
            if (n === 3) {
                n = 2;
